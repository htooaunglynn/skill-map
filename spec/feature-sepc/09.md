## 1. New Data Model

### 1.1 Overview

Four new top-level arrays are added to `skillmap.json`:

```
goals[]
  └── milestones[]          (ordered, belong to a goal)

practice_sessions[]         (linked to a goal, optionally to a milestone)

progress_notes[]            (linked to a goal, optionally to a milestone or session)
```

### 1.2 Add to `src/types/schema.ts`

```ts
// ── Goal ─────────────────────────────────────────────────────────────────────

export type GoalStatus = 'not_started' | 'in_progress' | 'completed' | 'archived';

export interface Goal {
  id: string;
  title: string;
  description: string;
  status: GoalStatus;
  target_date: string | null;       // ISO date string e.g. "2026-09-01"
  created_at: string;
  updated_at: string;
}

// ── Milestone ─────────────────────────────────────────────────────────────────

export type MilestoneStatus = 'not_started' | 'in_progress' | 'completed';

export interface Milestone {
  id: string;
  goal_id: string;                  // required — always belongs to a goal
  title: string;
  description: string;
  status: MilestoneStatus;
  order: number;                    // 1-based, user can reorder
  due_date: string | null;          // ISO date string or null
  completed_at: string | null;      // ISO datetime when status set to completed
  created_at: string;
  updated_at: string;
}

// ── Practice Session ──────────────────────────────────────────────────────────

export type SessionStatus = 'scheduled' | 'completed' | 'cancelled' | 'missed';

export interface PracticeSession {
  id: string;
  goal_id: string;                  // required
  milestone_id: string | null;      // optional
  title: string;
  notes: string;                    // plain markdown string
  status: SessionStatus;
  scheduled_at: string;             // ISO datetime e.g. "2026-06-20T14:00:00.000Z"
  duration_minutes: number;         // e.g. 60
  completed_at: string | null;      // ISO datetime when marked completed
  created_at: string;
  updated_at: string;
}

// ── Progress Note ─────────────────────────────────────────────────────────────

export interface ProgressNote {
  id: string;
  goal_id: string;                  // required
  milestone_id: string | null;      // optional
  session_id: string | null;        // optional — links note to a session
  title: string;
  content: string;                  // plain markdown string
  created_at: string;
  updated_at: string;
}
```

## 3. Derived Data Helpers

Create `src/lib/goalHelpers.ts`. These are pure functions — no side effects,
no context access. They receive arrays directly.

```ts
import type {
  Goal, Milestone, PracticeSession, ProgressNote, MilestoneStatus
} from '@/types/schema';

// ── Milestone helpers ─────────────────────────────────────────────────────────

/** All milestones for a goal, sorted by order ascending. */
export function getMilestonesForGoal(
  milestones: Milestone[],
  goalId: string
): Milestone[] {
  return milestones
    .filter(m => m.goal_id === goalId)
    .sort((a, b) => a.order - b.order);
}

/** Completion percentage for a goal based on completed milestones. */
export function getGoalProgress(
  milestones: Milestone[],
  goalId: string
): number {
  const all = getMilestonesForGoal(milestones, goalId);
  if (all.length === 0) return 0;
  const done = all.filter(m => m.status === 'completed').length;
  return Math.round((done / all.length) * 100);
}

/** Milestones that are overdue: due_date is in the past and not completed. */
export function getOverdueMilestones(milestones: Milestone[]): Milestone[] {
  const now = new Date();
  return milestones.filter(m => {
    if (m.status === 'completed') return false;
    if (!m.due_date) return false;
    return new Date(m.due_date) < now;
  });
}

// ── Session helpers ───────────────────────────────────────────────────────────

/** Sessions for a goal, optionally filtered by milestone. */
export function getSessionsForGoal(
  sessions: PracticeSession[],
  goalId: string,
  milestoneId?: string | null
): PracticeSession[] {
  return sessions.filter(s => {
    if (s.goal_id !== goalId) return false;
    if (milestoneId !== undefined) return s.milestone_id === milestoneId;
    return true;
  });
}

/** Sessions scheduled in the future, sorted by scheduled_at ascending. */
export function getUpcomingSessions(
  sessions: PracticeSession[]
): PracticeSession[] {
  const now = new Date();
  return sessions
    .filter(s => s.status === 'scheduled' && new Date(s.scheduled_at) > now)
    .sort((a, b) =>
      new Date(a.scheduled_at).getTime() - new Date(b.scheduled_at).getTime()
    );
}

/** Sessions where scheduled_at is in the past and status is still 'scheduled'. */
export function getMissedSessions(
  sessions: PracticeSession[]
): PracticeSession[] {
  const now = new Date();
  return sessions.filter(
    s => s.status === 'scheduled' && new Date(s.scheduled_at) < now
  );
}

/** Total completed practice minutes for a goal. */
export function getTotalPracticeMinutes(
  sessions: PracticeSession[],
  goalId: string
): number {
  return sessions
    .filter(s => s.goal_id === goalId && s.status === 'completed')
    .reduce((sum, s) => sum + s.duration_minutes, 0);
}

// ── Progress note helpers ─────────────────────────────────────────────────────

/** Notes for a goal, most recent first. */
export function getNotesForGoal(
  notes: ProgressNote[],
  goalId: string
): ProgressNote[] {
  return notes
    .filter(n => n.goal_id === goalId)
    .sort((a, b) =>
      new Date(b.created_at).getTime() - new Date(a.created_at).getTime()
    );
}

/** Most recently created notes across all goals, up to a limit. */
export function getRecentNotes(
  notes: ProgressNote[],
  limit = 5
): ProgressNote[] {
  return [...notes]
    .sort((a, b) =>
      new Date(b.created_at).getTime() - new Date(a.created_at).getTime()
    )
    .slice(0, limit);
}

// ── Search helpers ────────────────────────────────────────────────────────────

export function searchGoals(goals: Goal[], query: string): Goal[] {
  const q = query.toLowerCase().trim();
  if (!q) return goals;
  return goals.filter(
    g =>
      g.title.toLowerCase().includes(q) ||
      g.description.toLowerCase().includes(q)
  );
}

export function searchSessions(
  sessions: PracticeSession[],
  query: string
): PracticeSession[] {
  const q = query.toLowerCase().trim();
  if (!q) return sessions;
  return sessions.filter(
    s =>
      s.title.toLowerCase().includes(q) ||
      s.notes.toLowerCase().includes(q)
  );
}

export function searchNotes(
  notes: ProgressNote[],
  query: string
): ProgressNote[] {
  const q = query.toLowerCase().trim();
  if (!q) return notes;
  return notes.filter(
    n =>
      n.title.toLowerCase().includes(q) ||
      n.content.toLowerCase().includes(q)
  );
}
```

### 4.2 Create `src/lib/animations.ts`

All GSAP logic lives here. Components never import `gsap` directly.

```ts
import { gsap } from 'gsap';

export function shouldAnimate(): boolean {
  if (typeof window === 'undefined') return false;
  return !window.matchMedia('(prefers-reduced-motion: reduce)').matches;
}

// Page fade-in — call on the page root ref on mount
export function animatePageIn(el: HTMLElement) {
  if (!shouldAnimate()) return;
  gsap.fromTo(el,
    { opacity: 0, y: 16 },
    { opacity: 1, y: 0, duration: 0.4, ease: 'power2.out' }
  );
}

// Stagger a list of items — add data-animate-item to each child
export function animateListIn(els: NodeListOf<HTMLElement> | HTMLElement[]) {
  if (!shouldAnimate()) return;
  gsap.fromTo(els,
    { opacity: 0, y: 20 },
    { opacity: 1, y: 0, duration: 0.35, ease: 'power2.out', stagger: 0.07 }
  );
}

// New card sliding in after create
export function animateCardIn(el: HTMLElement) {
  if (!shouldAnimate()) return;
  gsap.fromTo(el,
    { opacity: 0, y: 24, scale: 0.97 },
    { opacity: 1, y: 0, scale: 1, duration: 0.35, ease: 'back.out(1.4)' }
  );
}

// Card exit before delete — await this before removing from state
export function animateCardOut(el: HTMLElement): Promise<void> {
  if (!shouldAnimate()) return Promise.resolve();
  return new Promise(resolve =>
    gsap.to(el, {
      opacity: 0, y: -12, scale: 0.96,
      duration: 0.25, ease: 'power2.in',
      onComplete: resolve,
    })
  );
}

// Dialog/modal entrance
export function animateModalIn(el: HTMLElement) {
  if (!shouldAnimate()) return;
  gsap.fromTo(el,
    { opacity: 0, scale: 0.95, y: 12 },
    { opacity: 1, scale: 1, y: 0, duration: 0.3, ease: 'power2.out' }
  );
}

// Progress bar count-up — returns tween so caller can kill() on unmount
export function animateProgress(
  setter: (v: number) => void,
  target: number,
  duration = 0.8
) {
  const obj = { value: 0 };
  return gsap.to(obj, {
    value: target,
    duration,
    ease: 'power1.out',
    onUpdate: () => setter(Math.round(obj.value)),
  });
}

// Milestone toggle flash — brief highlight on the row
export function animateMilestoneToggle(el: HTMLElement) {
  if (!shouldAnimate()) return;
  gsap.fromTo(el,
    { backgroundColor: '#d1fae5' },
    { backgroundColor: 'transparent', duration: 0.8, ease: 'power1.out' }
  );
}

// Status badge swap — scale bounce when status changes
export function animateStatusChange(el: HTMLElement) {
  if (!shouldAnimate()) return;
  gsap.fromTo(el,
    { scale: 0.85 },
    { scale: 1, duration: 0.3, ease: 'back.out(2)' }
  );
}

// Dashboard section stagger on mount
export function animateDashboardSections(
  els: NodeListOf<HTMLElement> | HTMLElement[]
) {
  if (!shouldAnimate()) return;
  gsap.fromTo(els,
    { opacity: 0, x: -16 },
    { opacity: 1, x: 0, duration: 0.3, ease: 'power1.out', stagger: 0.08 }
  );
}
```

### 4.3 Hooks

**`src/hooks/usePageAnimation.ts`**

```ts
'use client';
import { useEffect, useRef } from 'react';
import { animatePageIn } from '@/lib/animations';

export function usePageAnimation() {
  const ref = useRef<HTMLDivElement>(null);
  useEffect(() => { if (ref.current) animatePageIn(ref.current); }, []);
  return ref;
}
```

**`src/hooks/useListAnimation.ts`**

```ts
'use client';
import { useEffect, useRef } from 'react';
import { animateListIn } from '@/lib/animations';

export function useListAnimation(deps: unknown[] = []) {
  const ref = useRef<HTMLDivElement>(null);
  useEffect(() => {
    if (!ref.current) return;
    const items = ref.current.querySelectorAll<HTMLElement>('[data-animate-item]');
    if (items.length) animateListIn(items);
  }, deps); // eslint-disable-line react-hooks/exhaustive-deps
  return ref;
}
```

**`src/hooks/useProgressAnimation.ts`**

```ts
'use client';
import { useEffect, useState } from 'react';
import { animateProgress } from '@/lib/animations';

export function useProgressAnimation(target: number) {
  const [value, setValue] = useState(0);
  useEffect(() => {
    const t = animateProgress(setValue, target);
    return () => { t.kill(); };
  }, [target]);
  return value;
}
```

### 4.4 Animated Progress component

Create `src/components/ui/AnimatedProgress.tsx`:

```tsx
'use client';
import { Progress } from '@/components/ui/progress';
import { useProgressAnimation } from '@/hooks/useProgressAnimation';

interface Props {
  value: number;
  className?: string;
  showLabel?: boolean;
}

export function AnimatedProgress({ value, className, showLabel = false }: Props) {
  const display = useProgressAnimation(value);
  return (
    <div className={className}>
      <Progress value={display} />
      {showLabel && (
        <span className="text-xs text-muted-foreground mt-1 block">{display}%</span>
      )}
    </div>
  );
}
```

Replace all existing `<Progress value={...} />` usages in the codebase with
`<AnimatedProgress value={...} />`.

## 8. JavaScript-Powered Inline Interactions

These interactions update state without navigating away from the page:

| Interaction | Trigger | Animation |
|---|---|---|
| Milestone toggle complete | Checkbox click | `animateMilestoneToggle` on row |
| Session status update | Status dropdown change | `animateStatusChange` on badge |
| Goal filter tab change | Tab click | List re-renders with `animateListIn` |
